#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

void output(vector<int>);

int main(){
    vector<int> v(5, 999);
    output(v);
    
    v.push_back(5000);
    v.erase(v.begin() + 3);

    output(v);
    
    v.clear();
    
    
    v.resize(10);
    for(int i = 0; i < 10; i++){
        v[i] = 9-i;
    }
    
    output(v);
    
    sort(v.begin(), v.end());
    
    output(v);
    return 0;
}

void output(vector<int> v){
    for(int i = 0; i < v.size(); i++){
        cout<<v[i]<<" ";
    }
    cout<<endl;
}
