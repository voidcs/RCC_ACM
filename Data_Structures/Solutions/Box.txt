#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll t;
    cin>>t;
    while(t--){
        ll n;
        cin>>n;
        ll a[n];
        ll b[n];
        
        for(int i = 0; i < n; i++)
            cin>>a[i];
        set<ll>s;
        for(int i = 0; i < n; i++)
            s.insert(i+1);
        
        bool flag = 0;
        for(int i = 0; i < n; i++){
            if(s.count(a[i])){
                b[i] = a[i];
                s.erase(a[i]);
            }
            else{
                auto it = s.begin();
                if(*it > a[i]){
                    flag = 1;
                    break;
                }
                b[i] = *it;
                s.erase(*it); 
            }
        }
        if(flag){
            cout<<"-1\n";
        }
        else{
            for(int i = 0; i < n; i++){
                cout<<b[i]<<" ";
            }
            cout<<endl;
        }
    }
    return 0;
}
